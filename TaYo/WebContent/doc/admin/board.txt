
SELECT *
FROM    (SELECT rownum r, repboard.*
	FROM repboard
	START WITH parent_seq = 0
	CONNECT BY PRIOR board_seq = parent_seq
	ORDER SIBLINGS BY board_seq DESC)
WHERE r BETWEEN 1 AND 10;
